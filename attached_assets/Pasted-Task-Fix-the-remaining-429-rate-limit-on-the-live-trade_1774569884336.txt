import { NextRequest, NextResponse } from 'next/server';

const KEY1 = "54385120-28ac-4baa-9774-3f7ba8ccd656";
const KEY2 = "8ffc1afb-b4e5-494e-852e-f80ec0f5033e";

let currentKeyIndex = 0;

export async function POST(request: NextRequest) {
  try {
    const { wallet } = await request.json();
    if (!wallet) return NextResponse.json({ error: "Wallet required" }, { status: 400 });

    const keys = [KEY1, KEY2];
    let lastError = null;

    for (let attempt = 0; attempt < 2; attempt++) {
      const key = keys[currentKeyIndex];
      currentKeyIndex = (currentKeyIndex + 1) % 2;  // round-robin

      const url = `https://api-mainnet.helius-rpc.com/v0/addresses/${wallet}/transactions?api-key=${key}&type=SWAP&limit=50`;

      const res = await fetch(url, { method: "GET" });

      if (res.ok) {
        const data = await res.json();
        return NextResponse.json(data);
      }

      if (res.status !== 429) {
        lastError = res.status;
        break;
      }
      // If 429, try the other key
      await new Promise(r => setTimeout(r, 300)); // small delay
    }

    return NextResponse.json({ error: "Rate limit or Helius error" }, { status: lastError || 429 });
  } catch (err) {
    return NextResponse.json({ error: "Server error" }, { status: 500 });
  }
}
